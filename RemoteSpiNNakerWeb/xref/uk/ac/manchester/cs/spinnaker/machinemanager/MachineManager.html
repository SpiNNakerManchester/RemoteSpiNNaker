17  package uk.ac.manchester.cs.spinnaker.machinemanager;
18  
19  import java.util.List;
20  
21  import uk.ac.manchester.cs.spinnaker.machine.ChipCoordinates;
22  import uk.ac.manchester.cs.spinnaker.machine.SpinnakerMachine;
23  
24  /**
25   * A service for managing SpiNNaker boards in a machine.
26   */
27  public interface MachineManager extends AutoCloseable {
28      /**
29       * Gets the machines that this manager allocates from.
30       *
31       * @return collection of machines
32       */
33      List<SpinnakerMachine> getMachines();
34  
35      /**
36       * Get the next available machine of a given size.
37       *
38       * @param nBoards
39       *            The (minimum) number of boards that the machine needs to have.
40       * @return a machine.
41       */
42      SpinnakerMachine getNextAvailableMachine(int nBoards);
43  
44      /**
45       * Test if a specific machine is available.
46       *
47       * @param machine
48       *            The machine handle
49       * @return true if the machine is available.
50       */
51      boolean isMachineAvailable(SpinnakerMachine machine);
52  
53      /**
54       * Wait for the machine's availability to change.
55       *
56       * @param machine
57       *            The machine handle
58       * @param waitTime
59       *            Maximum wait time (in milliseconds)
60       * @return Whether the machine state has changed.
61       */
62      boolean waitForMachineStateChange(SpinnakerMachine machine, int waitTime);
63  
64      /**
65       * Release an allocated machine.
66       *
67       * @param machine
68       *            The machine handle
69       */
70      void releaseMachine(SpinnakerMachine machine);
71  
72      /**
73       * Turn a machine on or off.
74       *
75       * @param machine
76       *            The machine handle
77       * @param powerOn
78       *            True to power a machine on, false to turn it off.
79       */
80      void setMachinePower(SpinnakerMachine machine, boolean powerOn);
81  
82      /**
83       * Find a chip on a machine.
84       *
85       * @param machine
86       *            The machine handle
87       * @param x
88       *            The virtual X coordinate of the chip
89       * @param y
90       *            The virtual Y coordinate of the chip
91       * @return The chip location description
92       */
93      ChipCoordinates getChipCoordinates(SpinnakerMachine machine, int x, int y);
94  }
