44  /**
45   * Manufactures clients for talking to other machines. This class does wicked
46   * things, but it is because users keep insisting on getting their security
47   * wrong. How difficult is it to just use LetsEncrypt? Apparently very because
48   * users instead stand up impossible-to-trust self-signed certificates! The
49   * infrastructure required to allow this to work securely requires a lot of
50   * interface changes (they'd have to supply the server's public certificate
51   * chain as part of the job submission) and that's just awkward.
52   */
53  public abstract class RestClientUtils {
54  
55      /**
56       * Stop instances being created.
57       */
58      private RestClientUtils() {
59      } // No instances, please, we're British!
60  
61      /**
62       * The security protocol that is requested for a secure connection.
63       */
64      public static final String SECURE_PROTOCOL = "TLS";
65      /**
66       * The maximum total connections to allow.
67       */
68      private static final int MAX_CONNECTIONS = 2000;
69      /**
70       * The maximum total connections per route.
71       */
72      private static final int MAX_CONNECTIONS_PER_ROUTE = 200;
73  
74      /**
75       * The timeout of socket operation in seconds.
76       */
77      private static final long TIMEOUT = 60;
78  
79      /**
80       * Manufacture a client.
81       *
82       * @param url
83       *            What this client talks to.
84       * @return the client
85       */
86      protected static ResteasyClient createRestClient(final URL url) {
87          try {
88              // Create and return a client
89              final ResteasyClient client = new ResteasyClientBuilder()
90                      .connectTimeout(TIMEOUT, TimeUnit.SECONDS)
91                      .readTimeout(TIMEOUT, TimeUnit.SECONDS)
92                      .connectionPoolSize(MAX_CONNECTIONS)
93                      .maxPooledPerRoute(MAX_CONNECTIONS_PER_ROUTE)
94                      .hostnameVerification(HostnameVerificationPolicy.ANY)
95                      .sslContext(getSSLContext())
96                      .build();
97              client.register(new ErrorCaptureResponseFilter());
98              return client;
99          } catch (NoSuchAlgorithmException | KeyManagementException
100                 | KeyStoreException e) {
101             throw new RuntimeException("Unexpectedly broken security", e);
102         }
103     }
104 
105     /**
106      * Create an SSL context with appropriate key store and management.
107      *
108      * @return The created context
109      * @throws NoSuchAlgorithmException If the key store can't be read
110      * @throws KeyStoreException If the key store can't be read
111      * @throws KeyManagementException If the key store can't be read
112      */
113     private static SSLContext getSSLContext() throws NoSuchAlgorithmException,
114             KeyStoreException, KeyManagementException {
115         SSLContextBuilder builder = new SSLContextBuilder();
116         builder.loadTrustMaterial(new TrustSelfSignedStrategy());
117         String trustStore = System.getProperty(
118                 "remotespinnaker.keystore", null);
119         if (trustStore != null) {
120             String password = System.getProperty(
121                     "remotespinnaker.keystore.password", "");
122             try {
123                 builder.loadTrustMaterial(new File(trustStore),
124                         password.toCharArray());
125             } catch (IOException | CertificateException e) {
126                 throw new RuntimeException(
127                         "Unexpected error loading certificates", e);
128             }
129         }
130         return builder.build();
131     }
132 
133     /**
134      * Create a REST client proxy for a class to the given URL.
135      *
136      * @param <T>
137      *            The type of interface to proxy
138      * @param url
139      *            The URL of the REST service
140      * @param authorizationHeader
141      *            The authorisation header to provide
142      * @param clazz
143      *            The interface to proxy
144      * @param providers
145      *            The objects to register with the underlying client
146      * @return The proxy instance
147      */
148     public static <T> T createClient(final URL url,
149             final String authorizationHeader,
150             final Class<T> clazz, final Object... providers) {
151         final ResteasyClient client = createRestClient(url);
152         for (final Object provider : providers) {
153             client.register(provider);
154         }
155         if (providers.length == 0) {
156             client.register(new JacksonJsonProvider());
157         }
158         return client.target(url.toString()).register(
159                 new ClientRequestFilter() {
160 
161             @Override
162             public void filter(final ClientRequestContext context)
163                     throws IOException {
164                 context.getHeaders().add(WWW_AUTH_RESP, authorizationHeader);
165             }
166         }).proxy(clazz);
167     }
168 
169     /**
170      * Create a new REST client with BASIC authentication.
171      *
172      * @param <T>
173      *            The type of interface to proxy
174      * @param url
175      *            The URL of the REST service
176      * @param username
177      *            The user name of the user accessing the service
178      * @param password
179      *            The password for authentication
180      * @param clazz
181      *            The interface to proxy
182      * @param providers
183      *            The objects to register with the underlying client
184      * @return The proxy instance
185      */
186     public static <T> T createBasicClient(final URL url, final String username,
187             final String password, final Class<T> clazz,
188             final Object... providers) {
189         String userPass = Base64.encodeBase64String(
190                 EncodingUtils.getAsciiBytes(username + ":" + password));
191         String authHeader = "Basic " + userPass;
192         return createClient(url, authHeader, clazz, providers);
193     }
194 
195     /**
196      * Create a new REST client with APIKey authentication.
197      *
198      * @param <T>
199      *            The type of interface to proxy
200      * @param url
201      *            The URL of the REST service
202      * @param username
203      *            The user name of the user accessing the service
204      * @param apiKey
205      *            The key to use to authenticate
206      * @param clazz
207      *            The interface to proxy
208      * @param providers
209      *            The objects to register with the underlying client
210      * @return The proxy instance
211      */
212     public static <T> T createApiKeyClient(final URL url, final String username,
213             final String apiKey, final Class<T> clazz,
214             final Object... providers) {
215         return createClient(url, "ApiKey " + username + ":" + apiKey,
216                 clazz, providers);
217     }
218 
219     /**
220      * Create a new REST client with Bearer authentication.
221      *
222      * @param <T>
223      *            The type of interface to proxy
224      * @param url
225      *            The URL of the REST service
226      * @param token
227      *            The Bearer token for authentication
228      * @param clazz
229      *            The interface to proxy
230      * @param providers
231      *            The objects to register with the underlying client
232      * @return The proxy instance
233      */
234     public static <T> T createBearerClient(final URL url, final String token,
235             final Class<T> clazz, final Object... providers) {
236         return createClient(url, "Bearer " + token, clazz, providers);
237     }
