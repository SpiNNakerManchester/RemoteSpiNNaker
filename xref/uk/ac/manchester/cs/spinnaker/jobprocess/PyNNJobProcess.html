178     /**
179      * A filter to remove files with ignored instructions.
180      *
181      * @return The file filter
182      */
183     private static IOFileFilter fileFilter() {
184         return new AbstractFileFilter() {
185             @Override
186             public boolean accept(final File file) {
187                 return !IGNORED_EXTENSIONS
188                         .contains(getExtension(file.getName()));
189             }
190         };
191     }
192 
193     /**
194      * A filter to remove ignored directories.
195      *
196      * @return The directory filter
197      */
198     private static IOFileFilter directoryFilter() {
199         return new AbstractFileFilter() {
200             @Override
201             public boolean accept(final File file) {
202                 return !IGNORED_DIRECTORIES.contains(file.getName());
203             }
204         };
205     }

207     /**
208      * Executes the process.
209      */
210     @Override
211     public void execute(final String machineUrl, final SpinnakerMachine machine,
212             final PyNNJobParameters parameters, final LogWriter logWriter) {
213         try {
214             status = Running;
215             workingDirectory = new File(parameters.getWorkingDirectory());
216 
217             // Run the setup
218             final int setupValue = runSetup(parameters, logWriter);
219             if (setupValue != 0) {
220                 throw new Exception("Setup exited with non-zero error code + ("
221                         + setupValue + ")");
222             }
223 
224             // Create a spynnaker config file
225             final File cfgFile = new File(workingDirectory, "spynnaker.cfg");
226 
227             // Add the details of the machine
228             final ConfigParser parser = new ConfigParser();
229             if (cfgFile.exists()) {
230                 parser.read(cfgFile);
231             }
232 
233             if (!parser.hasSection(SECTION)) {
234                 parser.addSection(SECTION);
235             }
236             if (machine != null) {
237                 parser.set(SECTION, "machineName", machine.getMachineName());
238                 parser.set(SECTION, "version", machine.getVersion());
239                 parser.set(SECTION, "width", machine.getWidth());
240                 parser.set(SECTION, "height", machine.getHeight());
241                 final String bmpDetails = machine.getBmpDetails();
242                 if (bmpDetails != null) {
243                     parser.set(SECTION, "bmp_names", bmpDetails);
244                 }
245             } else {
246                 parser.set(SECTION, "remote_spinnaker_url", machineUrl);
247             }
248             parser.write(cfgFile);
249 
250             // Keep existing files to compare to later
251             final Set<File> existingFiles = gatherFiles(workingDirectory);
252 
253             // Execute the program
254             final int exitValue = runSubprocess(parameters, logWriter);
255 
256             // Get the provenance data
257             gatherProvenance(workingDirectory);
258 
259             // Get any output files
260             final Set<File> allFiles = gatherFiles(workingDirectory);
261             for (final File file : allFiles) {
262                 if (!existingFiles.contains(file)) {
263                     outputs.add(file);
264                 }
265             }
266 
267             // If the exit is an error, mark an error
268             if (exitValue >= MIN_SIGNAL_OFFSET) {
269                 // Useful to distinguish this case
270                 throw new Exception("Python exited with signal ("
271                         + (exitValue - MIN_SIGNAL_OFFSET) + ")");
272             }
273             if (exitValue != 0) {
274                 throw new Exception("Python exited with a non-zero code ("
275                         + exitValue + ")");
276             }
277             status = Finished;
278         } catch (final Throwable e) {
279             StringWriter stringWriter = new StringWriter();
280             PrintWriter printWriter = new PrintWriter(stringWriter);
281             e.printStackTrace(printWriter);
282             logWriter.append(stringWriter.toString());
283             e.printStackTrace();
284             error = e;
285             status = Error;
286         }
287     }

289     /**
290      * Run the setup process.
291      *
292      * @param parameters
293      *            The parameters to the setup process.
294      * @param logWriter
295      *            Where to send log messages.
296      * @return
297      *            The exit value of the process
298      * @throws IOException
299      *            If there was an error starting the process
300      * @throws InterruptedException
301      *            If the process was interrupted before return
302      */
303     private int runSetup(final PyNNJobParameters parameters,
304             final LogWriter logWriter)
305             throws IOException, InterruptedException {
306         final List<String> command = new ArrayList<>();
307         command.add(SETUP_RUNNER);
308         command.add(parameters.getSetupScript());
309 
310         // Build a process
311         final ProcessBuilder builder = new ProcessBuilder(command);
312         builder.directory(workingDirectory);
313         builder.redirectErrorStream(true);
314         ObjectMapper mapper = new ObjectMapper();
315         for (Entry<String, Object> entry
316                 : parameters.getHardwareConfiguration().entrySet()) {
317             String stringValue = null;
318             Object value = entry.getValue();
319             if (value instanceof String) {
320                 stringValue = (String) value;
321             } else {
322                 stringValue = mapper.writeValueAsString(value);
323             }
324             builder.environment().put(entry.getKey(), stringValue);
325         }
326         final Process process = builder.start();
327 
328         // Run a thread to gather the log
329         try (ReaderLogWriter logger =
330                 new ReaderLogWriter(process.getInputStream(), logWriter)) {
331             logger.start();
332 
333             // Wait for the process to finish
334             return process.waitFor();
335         }
336     }

338     /**
339      * How to actually run a subprocess.
340      *
341      * @param parameters
342      *            The parameters to the subprocess.
343      * @param logWriter
344      *            Where to send log messages.
345      * @return
346      *            The exit value of the process
347      * @throws IOException
348      *            If there was an error starting the process
349      * @throws InterruptedException
350      *            If the process was interrupted before return
351      */
352     private int runSubprocess(final PyNNJobParameters parameters,
353             final LogWriter logWriter)
354             throws IOException, InterruptedException {
355         final List<String> command = new ArrayList<>();
356         command.add(SUBPROCESS_RUNNER);
357 
358         final Matcher scriptMatcher =
359                 ARGUMENT_FINDER.matcher(parameters.getUserScript());
360         while (scriptMatcher.find()) {
361             command.add(
362                     scriptMatcher.group(1).replace("{system}", "spiNNaker"));
363         }
364 
365         final ProcessBuilder builder = new ProcessBuilder(command);
366         log("Running " + command + " in " + workingDirectory);
367         builder.directory(workingDirectory);
368         builder.redirectErrorStream(true);
369         final Process process = builder.start();
370 
371         // Run a thread to gather the log
372         try (ReaderLogWriter logger =
373                 new ReaderLogWriter(process.getInputStream(), logWriter)) {
374             logger.start();
375 
376             // Wait for the process to finish
377             return process.waitFor();
378         }
379     }

381     /**
382      * Enter some provenance into the provenance map.
383      *
384      * @param items
385      *            The items to insert.
386      * @param path
387      *            Where to insert these items relative to the current node, as a
388      *            string.
389      * @param pathList
390      *            Where to insert these items relative to the current node, as a
391      *            list.
392      */
393     private void putProvenanceInMap(final ProvenanceDataItems items,
394             final String path, final LinkedList<String> pathList) {
395 
396         // Create a path for this level in the tree
397         final String myPath = path + items.getName();
398         pathList.addLast(items.getName());
399 
400         // Add all nested items
401         for (final ProvenanceDataItems subItems : items
402                 .getProvenanceDataItems()) {
403             putProvenanceInMap(subItems, myPath + "/", pathList);
404         }
405 
406         // Add items from this level
407         for (final ProvenanceDataItem subItem : items.getProvenanceDataItem()) {
408             final String itemPath = myPath + "/" + subItem.getName();
409             pathList.addLast(subItem.getName());
410             for (final String item : PROVENANCE_ITEMS_TO_ADD) {
411                 if (itemPath.matches(item)) {
412                     provenance.add(new ProvenanceItem(
413                         new ArrayList<>(pathList), subItem.getValue()));
414                 }
415             }
416             pathList.removeLast();
417         }
418         pathList.removeLast();
419     }

421     /**
422      * Add the provenance contained in the files in the given directory.
423      *
424      * @param provenanceDirectory
425      *            Where to look for XML files.
426      * @throws JAXBException
427      *             If things go wrong in deserialisation.
428      * @throws IOException
429      *             If anything goes wrong with I/O.
430      */
431     private void addProvenance(final File provenanceDirectory)
432             throws IOException, JAXBException {
433 
434         // Get provenance data from files
435         for (final File file : provenanceDirectory.listFiles()) {
436 
437             // Only process XML files
438             if (file.getName().endsWith(".xml")) {
439                 final JAXBContext jaxbContext =
440                         JAXBContext.newInstance(ProvenanceDataItems.class);
441                 final Unmarshaller jaxbUnmarshaller =
442                         jaxbContext.createUnmarshaller();
443                 final Source source = new StreamSource(file);
444                 final ProvenanceDataItems items =
445                         (ProvenanceDataItems) jaxbUnmarshaller
446                                 .unmarshal(source);
447                 putProvenanceInMap(items, "", new LinkedList<String>());
448             }
449         }
450     }

452     /**
453      * Used for creating a ZIP of the provenance.
454      *
455      * @param reportsZip
456      *            Open handle to the ZIP being created.
457      * @param directory
458      *            Where to get provenance data from.
459      * @param path
460      *            The path within the ZIP.
461      * @throws IOException
462      *             If anything goes wrong with I/O.
463      * @throws JAXBException
464      *             If anything goes wrong with deserialisation of the XML.
465      */
466     private void zipProvenance(final ZipOutputStream reportsZip,
467             final File directory, final String path)
468             throws IOException, JAXBException {
469 
470         // Go through the report files and zip them up
471         final byte[] buffer = new byte[BUFFER_SIZE];
472         for (final File file : directory.listFiles()) {
473             if (file.isDirectory()) {
474                 zipProvenance(reportsZip, file, path + "/" + file.getName());
475 
476                 // If the directory is the provenance directory, process it
477                 if (file.getName().equals(PROVENANCE_DIRECTORY)) {
478                     addProvenance(file);
479                 }
480             } else {
481                 final ZipEntry entry =
482                         new ZipEntry(path + "/" + file.getName());
483                 reportsZip.putNextEntry(entry);
484                 try (FileInputStream in = new FileInputStream(file)) {
485                     int bytesRead = in.read(buffer);
486                     while (bytesRead >= 0) {
487                         reportsZip.write(buffer, 0, bytesRead);
488                         bytesRead = in.read(buffer);
489                     }
490                 }
491             }
492         }
493     }

495     /**
496      * Gather the provenance information from the job's reports directory.
497      *
498      * @param workingDirectoryParam
499      *            The job's working directory.
500      * @throws IOException
501      *             If anything goes wrong with I/O.
502      * @throws JAXBException
503      *             If anything goes wrong with deserialisation of XML.
504      */
505     private void gatherProvenance(final File workingDirectoryParam)
506             throws IOException, JAXBException {
507 
508         // Find the reports folder
509         final File reportsFolder = new File(workingDirectoryParam, "reports");
510         if (reportsFolder.isDirectory()) {
511 
512             // Create a zip file of the reports
513             try (ZipOutputStream reportsZip =
514                     new ZipOutputStream(new FileOutputStream(
515                             new File(workingDirectoryParam, "reports.zip")))) {
516                 // Gather items into the reports zip, keeping an eye out for
517                 // the "provenance data" folder
518                 zipProvenance(reportsZip, reportsFolder, "reports");
519             }
520         }
521     }

563     /**
564      * Thread for copying a {@link Reader} to a {@link LogWriter}.
565      */
566     class ReaderLogWriter extends Thread implements AutoCloseable {
567 
568         /**
569          * The reader to read from.
570          */
571         private final BufferedReader reader;
572 
573         /**
574          * The writer to write to.
575          */
576         private final LogWriter writer;
577 
578         /**
579          * True when running, False to stop.
580          */
581         private boolean running;
582 
583         /**
584          * Creates a new ReaderLogWriter with another reader.
585          *
586          * @param readerParam
587          *            The reader to read from
588          * @param writerParam
589          *            The writer to write to
590          */
591         ReaderLogWriter(
592                 final Reader readerParam, final LogWriter writerParam) {
593             super(threadGroup, "Reader Log Writer");
594             requireNonNull(readerParam);
595             if (readerParam instanceof BufferedReader) {
596                 this.reader = (BufferedReader) readerParam;
597             } else {
598                 this.reader = new BufferedReader(readerParam);
599             }
600             this.writer = requireNonNull(writerParam);
601             setDaemon(true);
602         }
603 
604         /**
605          * Creates a new ReaderLogWriter with an input stream. This will be
606          * treated as a text stream using the system encoding.
607          *
608          * @param input
609          *            The input stream to read from.
610          * @param writerParam
611          *            The writer to write to.
612          */
613         ReaderLogWriter(final InputStream input,
614                 final LogWriter writerParam) {
615             this(new InputStreamReader(input), writerParam);
616         }
617 
618         @Override
619         public void run() {
620             try {
621                 copyStream();
622             } catch (IOException | RuntimeException e) {
623                 return;
624             } finally {
625                 synchronized (this) {
626                     running = false;
627                     notifyAll();
628                 }
629             }
630         }
631 
632         @Override
633         public void start() {
634             running = true;
635             super.start();
636         }
637 
638         /**
639          * Perform the copying of the stream.
640          *
641          * @throws IOException If there is an error copying
642          */
643         private void copyStream() throws IOException {
644             while (!interrupted()) {
645                 final String line = reader.readLine();
646                 if (line == null) {
647                     return;
648                 }
649                 writer.append(line + "\n");
650             }
651         }
652 
653         /**
654          * Closes the reader/writer.
655          */
656         @Override
657         public void close() {
658             log("Waiting for log writer to exit...");
659 
660             synchronized (this) {
661                 try {
662                     while (running) {
663                         wait();
664                     }
665                 } catch (final InterruptedException e) {
666                     // Does Nothing
667                 }
668             }
669 
670             log("Log writer has exited");
671             try {
672                 reader.close();
673             } catch (IOException | RuntimeException e) {
674                 // Do nothing
675             }
676             ThreadUtils.sleep(FINALIZATION_DELAY);
677         }
678     }
