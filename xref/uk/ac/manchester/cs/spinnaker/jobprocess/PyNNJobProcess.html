180     /**
181      * A filter to remove files with ignored instructions.
182      *
183      * @return The file filter
184      */
185     private static IOFileFilter fileFilter() {
186         return new AbstractFileFilter() {
187             @Override
188             public boolean accept(final File file) {
189                 return !IGNORED_EXTENSIONS
190                         .contains(getExtension(file.getName()));
191             }
192         };
193     }

195     /**
196      * A filter to remove ignored directories.
197      *
198      * @return The directory filter
199      */
200     private static IOFileFilter directoryFilter() {
201         return new AbstractFileFilter() {
202             @Override
203             public boolean accept(final File file) {
204                 return !IGNORED_DIRECTORIES.contains(file.getName());
205             }
206         };

209     /**
210      * Executes the process.
211      */
212     @Override
213     public void execute(final String machineUrl, final SpinnakerMachine machine,
214             final PyNNJobParameters parameters, final LogWriter logWriter) {
215         try {
216             status = Running;
217             workingDirectory = new File(parameters.getWorkingDirectory());
218 
219             // Run the setup
220             final int setupValue = runSetup(parameters, logWriter);
221             if (setupValue != 0) {
222                 throw new Exception("Setup exited with non-zero error code + ("
223                         + setupValue + ")");
224             }
225 
226             // Create a spynnaker config file
227             final File cfgFile = new File(workingDirectory, "spynnaker.cfg");
228 
229             // Add the details of the machine
230             final Ini ini = new Ini();
231             final Config config = ini.getConfig();
232             config.setEscape(false);
233             config.setLowerCaseSection(false);
234             config.setLowerCaseOption(false);
235             if (cfgFile.exists()) {
236                 ini.load(cfgFile);
237             }
238 
239             final Section section;
240             if (!ini.containsKey(SECTION)) {
241                 section = ini.add(SECTION);
242             } else {
243                 section = ini.get(SECTION);
244             }
245             if (machine != null) {
246                 section.put("machine_name", machine.getMachineName());
247                 section.put("version", machine.getVersion());
248                 final String bmpDetails = machine.getBmpDetails();
249                 if (bmpDetails != null) {
250                     section.put("bmp_names", bmpDetails);
251                 }
252             } else {
253                 section.put("remote_spinnaker_url", machineUrl);
254             }
255             ini.store(cfgFile);
256 
257             // Keep existing files to compare to later
258             final Set<File> existingFiles = gatherFiles(workingDirectory);
259 
260             // Execute the program
261             final int exitValue = runSubprocess(parameters, logWriter);
262 
263             // Get the provenance data
264             gatherProvenance(workingDirectory);
265 
266             // Get any output files
267             final Set<File> allFiles = gatherFiles(workingDirectory);
268             for (final File file : allFiles) {
269                 if (!existingFiles.contains(file)) {
270                     outputs.add(file);
271                 }
272             }
273 
274             // If the exit is an error, mark an error
275             if (exitValue >= MIN_SIGNAL_OFFSET) {
276                 // Useful to distinguish this case
277                 throw new Exception("Python exited with signal ("
278                         + (exitValue - MIN_SIGNAL_OFFSET) + ")");
279             }
280             if (exitValue != 0) {
281                 throw new Exception("Python exited with a non-zero code ("
282                         + exitValue + ")");
283             }
284             status = Finished;
285         } catch (final Throwable e) {
286             StringWriter stringWriter = new StringWriter();
287             PrintWriter printWriter = new PrintWriter(stringWriter);
288             e.printStackTrace(printWriter);
289             logWriter.append(stringWriter.toString());
290             e.printStackTrace();
291             error = e;
292             status = Error;
293         }
294     }

296     /**
297      * Run the setup process.
298      *
299      * @param parameters
300      *            The parameters to the setup process.
301      * @param logWriter
302      *            Where to send log messages.
303      * @return
304      *            The exit value of the process
305      * @throws IOException
306      *            If there was an error starting the process
307      * @throws InterruptedException
308      *            If the process was interrupted before return
309      */
310     private int runSetup(final PyNNJobParameters parameters,
311             final LogWriter logWriter)
312             throws IOException, InterruptedException {
313         final List<String> command = new ArrayList<>();
314         command.add(SETUP_RUNNER);
315         command.add(parameters.getSetupScript());
316 
317         // Build a process
318         final ProcessBuilder builder = new ProcessBuilder(command);
319         builder.directory(workingDirectory);
320         builder.redirectErrorStream(true);
321         ObjectMapper mapper = new ObjectMapper();
322         for (Entry<String, Object> entry
323                 : parameters.getHardwareConfiguration().entrySet()) {
324             String stringValue = null;
325             Object value = entry.getValue();
326             if (value instanceof String) {
327                 stringValue = (String) value;
328             } else {
329                 stringValue = mapper.writeValueAsString(value);
330             }
331             builder.environment().put(entry.getKey(), stringValue);
332         }
333         final Process process = builder.start();
334 
335         // Run a thread to gather the log
336         try (ReaderLogWriter logger =
337                 new ReaderLogWriter(process.getInputStream(), logWriter)) {
338             logger.start();
339 
340             // Wait for the process to finish
341             return process.waitFor();
342         }
343     }

345     /**
346      * How to actually run a subprocess.
347      *
348      * @param parameters
349      *            The parameters to the subprocess.
350      * @param logWriter
351      *            Where to send log messages.
352      * @return
353      *            The exit value of the process
354      * @throws IOException
355      *            If there was an error starting the process
356      * @throws InterruptedException
357      *            If the process was interrupted before return
358      */
359     private int runSubprocess(final PyNNJobParameters parameters,
360             final LogWriter logWriter)
361             throws IOException, InterruptedException {
362         final List<String> command = new ArrayList<>();
363         command.add(SUBPROCESS_RUNNER);
364 
365         final Matcher scriptMatcher =
366                 ARGUMENT_FINDER.matcher(parameters.getUserScript());
367         while (scriptMatcher.find()) {
368             command.add(
369                     scriptMatcher.group(1).replace("{system}", "spiNNaker"));
370         }
371 
372         final ProcessBuilder builder = new ProcessBuilder(command);
373         log("Running " + command + " in " + workingDirectory);
374         builder.directory(workingDirectory);
375         builder.redirectErrorStream(true);
376         final Process process = builder.start();
377 
378         // Run a thread to gather the log
379         try (ReaderLogWriter logger =
380                 new ReaderLogWriter(process.getInputStream(), logWriter)) {
381             logger.start();
382 
383             // Wait for the process to finish
384             return process.waitFor();
385         }
386     }

388     /**
389      * Enter some provenance into the provenance map.
390      *
391      * @param items
392      *            The items to insert.
393      * @param path
394      *            Where to insert these items relative to the current node, as a
395      *            string.
396      * @param pathList
397      *            Where to insert these items relative to the current node, as a
398      *            list.
399      */
400     private void putProvenanceInMap(final ProvenanceDataItems items,
401             final String path, final LinkedList<String> pathList) {
402 
403         // Create a path for this level in the tree
404         final String myPath = path + items.getName();
405         pathList.addLast(items.getName());
406 
407         // Add all nested items
408         for (final ProvenanceDataItems subItems : items
409                 .getProvenanceDataItems()) {
410             putProvenanceInMap(subItems, myPath + "/", pathList);
411         }
412 
413         // Add items from this level
414         for (final ProvenanceDataItem subItem : items.getProvenanceDataItem()) {
415             final String itemPath = myPath + "/" + subItem.getName();
416             pathList.addLast(subItem.getName());
417             for (final String item : PROVENANCE_ITEMS_TO_ADD) {
418                 if (itemPath.matches(item)) {
419                     provenance.add(new ProvenanceItem(
420                         new ArrayList<>(pathList), subItem.getValue()));
421                 }
422             }
423             pathList.removeLast();
424         }
425         pathList.removeLast();
426     }

428     /**
429      * Add the provenance contained in the files in the given directory.
430      *
431      * @param provenanceDirectory
432      *            Where to look for XML files.
433      * @throws JAXBException
434      *             If things go wrong in deserialisation.
435      * @throws IOException
436      *             If anything goes wrong with I/O.
437      */
438     private void addProvenance(final File provenanceDirectory)
439             throws IOException, JAXBException {
440 
441         // Get provenance data from files
442         for (final File file : provenanceDirectory.listFiles()) {
443 
444             // Only process XML files
445             if (file.getName().endsWith(".xml")) {
446                 final JAXBContext jaxbContext =
447                         JAXBContext.newInstance(ProvenanceDataItems.class);
448                 final Unmarshaller jaxbUnmarshaller =
449                         jaxbContext.createUnmarshaller();
450                 final Source source = new StreamSource(file);
451                 final ProvenanceDataItems items =
452                         (ProvenanceDataItems) jaxbUnmarshaller
453                                 .unmarshal(source);
454                 putProvenanceInMap(items, "", new LinkedList<String>());
455             }
456         }
457     }

459     /**
460      * Used for creating a ZIP of the provenance.
461      *
462      * @param reportsZip
463      *            Open handle to the ZIP being created.
464      * @param directory
465      *            Where to get provenance data from.
466      * @param path
467      *            The path within the ZIP.
468      * @throws IOException
469      *             If anything goes wrong with I/O.
470      * @throws JAXBException
471      *             If anything goes wrong with deserialisation of the XML.
472      */
473     private void zipProvenance(final ZipOutputStream reportsZip,
474             final File directory, final String path)
475             throws IOException, JAXBException {
476 
477         // Go through the report files and zip them up
478         final byte[] buffer = new byte[BUFFER_SIZE];
479         for (final File file : directory.listFiles()) {
480             if (file.isDirectory()) {
481                 zipProvenance(reportsZip, file, path + "/" + file.getName());
482 
483                 // If the directory is the provenance directory, process it
484                 if (file.getName().equals(PROVENANCE_DIRECTORY)) {
485                     addProvenance(file);
486                 }
487             } else {
488                 final ZipEntry entry =
489                         new ZipEntry(path + "/" + file.getName());
490                 reportsZip.putNextEntry(entry);
491                 try (FileInputStream in = new FileInputStream(file)) {
492                     int bytesRead = in.read(buffer);
493                     while (bytesRead >= 0) {
494                         reportsZip.write(buffer, 0, bytesRead);
495                         bytesRead = in.read(buffer);
496                     }
497                 }
498             }
499         }
500     }

502     /**
503      * Gather the provenance information from the job's reports directory.
504      *
505      * @param workingDirectoryParam
506      *            The job's working directory.
507      * @throws IOException
508      *             If anything goes wrong with I/O.
509      * @throws JAXBException
510      *             If anything goes wrong with deserialisation of XML.
511      */
512     private void gatherProvenance(final File workingDirectoryParam)
513             throws IOException, JAXBException {
514 
515         // Find the reports folder
516         final File reportsFolder = new File(workingDirectoryParam, "reports");
517         if (reportsFolder.isDirectory()) {
518 
519             // Create a zip file of the reports
520             try (ZipOutputStream reportsZip =
521                     new ZipOutputStream(new FileOutputStream(
522                             new File(workingDirectoryParam, "reports.zip")))) {
523                 // Gather items into the reports zip, keeping an eye out for
524                 // the "provenance data" folder
525                 zipProvenance(reportsZip, reportsFolder, "reports");
526             }
527         }
528     }

570     /**
571      * Thread for copying a {@link Reader} to a {@link LogWriter}.
572      */
573     class ReaderLogWriter extends Thread implements AutoCloseable {
574 
575         /**
576          * The reader to read from.
577          */
578         private final BufferedReader reader;
579 
580         /**
581          * The writer to write to.
582          */
583         private final LogWriter writer;
584 
585         /**
586          * True when running, False to stop.
587          */
588         private boolean running;
589 
590         /**
591          * Creates a new ReaderLogWriter with another reader.
592          *
593          * @param readerParam
594          *            The reader to read from
595          * @param writerParam
596          *            The writer to write to
597          */
598         ReaderLogWriter(
599                 final Reader readerParam, final LogWriter writerParam) {
600             super(threadGroup, "Reader Log Writer");
601             requireNonNull(readerParam);
602             if (readerParam instanceof BufferedReader) {
603                 this.reader = (BufferedReader) readerParam;
604             } else {
605                 this.reader = new BufferedReader(readerParam);
606             }
607             this.writer = requireNonNull(writerParam);
608             setDaemon(true);
609         }
610 
611         /**
612          * Creates a new ReaderLogWriter with an input stream. This will be
613          * treated as a text stream using the system encoding.
614          *
615          * @param input
616          *            The input stream to read from.
617          * @param writerParam
618          *            The writer to write to.
619          */
620         ReaderLogWriter(final InputStream input,
621                 final LogWriter writerParam) {
622             this(new InputStreamReader(input), writerParam);
623         }
624 
625         @Override
626         public void run() {
627             try {
628                 copyStream();
629             } catch (IOException | RuntimeException e) {
630                 return;
631             } finally {
632                 synchronized (this) {
633                     running = false;
634                     notifyAll();
635                 }
636             }
637         }
638 
639         @Override
640         public void start() {
641             running = true;
642             super.start();
643         }
644 
645         /**
646          * Perform the copying of the stream.
647          *
648          * @throws IOException If there is an error copying
649          */
650         private void copyStream() throws IOException {
651             while (!interrupted()) {
652                 final String line = reader.readLine();
653                 if (line == null) {
654                     return;
655                 }
656                 writer.append(line + "\n");
657             }
658         }
659 
660         /**
661          * Closes the reader/writer.
662          */
663         @Override
664         public void close() {
665             log("Waiting for log writer to exit...");
666 
667             synchronized (this) {
668                 try {
669                     while (running) {
670                         wait();
671                     }
672                 } catch (final InterruptedException e) {
673                     // Does Nothing
674                 }
675             }
676 
677             log("Log writer has exited");
678             try {
679                 reader.close();
680             } catch (IOException | RuntimeException e) {
681                 // Do nothing
682             }
683             ThreadUtils.sleep(FINALIZATION_DELAY);
684         }
685     }
