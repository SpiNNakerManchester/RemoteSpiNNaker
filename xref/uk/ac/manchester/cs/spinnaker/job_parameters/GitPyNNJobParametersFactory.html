18  import static java.util.Objects.nonNull;
19  import static org.eclipse.jgit.api.Git.cloneRepository;
20  
21  import java.io.File;
22  import java.net.URISyntaxException;
23  
24  import org.eclipse.jgit.api.errors.GitAPIException;
25  import org.eclipse.jgit.api.errors.InvalidRemoteException;
26  import org.eclipse.jgit.api.errors.TransportException;
27  import org.eclipse.jgit.transport.URIish;
28  import org.eclipse.jgit.transport.UsernamePasswordCredentialsProvider;
29  
30  import uk.ac.manchester.cs.spinnaker.job.JobParameters;
31  import uk.ac.manchester.cs.spinnaker.job.nmpi.Job;
32  import uk.ac.manchester.cs.spinnaker.job.pynn.PyNNJobParameters;

34  /**
35   * A {@link JobParametersFactory} that downloads a PyNN job from git. The git
36   * repository must be world-readable, or sufficient credentials must be present
37   * in the URL.
38   */
39  class GitPyNNJobParametersFactory extends JobParametersFactory {
40      @Override
41      public JobParameters getJobParameters(final Job job,
42              final File workingDirectory, final String setupScript)
43              throws UnsupportedJobException, JobParametersFactoryException {
44          // Test that there is a URL
45          final var jobCodeLocation = job.getCode().trim();
46          if (!jobCodeLocation.startsWith("http://")
47                  && !jobCodeLocation.startsWith("https://")) {
48              throw new UnsupportedJobException();
49          }
50  
51          // Try to get the repository
52          try {
53              return constructParameters(job, workingDirectory, jobCodeLocation,
54                      setupScript);
55          } catch (final InvalidRemoteException e) {
56              throw new JobParametersFactoryException("Remote is not valid", e);
57          } catch (final TransportException e) {
58              throw new JobParametersFactoryException("Transport failed", e);
59          } catch (final GitAPIException e) {
60              throw new JobParametersFactoryException("Error using Git", e);
61          } catch (final Throwable e) {
62              throw new JobParametersFactoryException(
63                      "General error getting git repository", e);
64          }
65      }
66  
67      /**
68       * Constructs the parameters by checking out the git repository.
69       *
70       * @param job The job to get the parameters of
71       * @param workingDirectory The directory where the job should be run
72       * @param experimentDescription The git URL
73       * @param setupScript The setup script to run
74       * @return The constructed parameters
75       * @throws GitAPIException If there is a general problem using Git
76       * @throws InvalidRemoteException If there is a problem with the repository
77       * @throws TransportException If there is a problem in communication
78       * @throws URISyntaxException If the URI syntax is incorrect
79       */
80      private JobParameters constructParameters(final Job job,
81              final File workingDirectory, final String experimentDescription,
82              final String setupScript)
83              throws GitAPIException, InvalidRemoteException, TransportException,
84              URISyntaxException {
85          final var clone = cloneRepository();
86          var urish = new URIish(experimentDescription);
87          if (nonNull(urish.getUser())) {
88              var pass = urish.getPass();
89              if (nonNull(pass)) {
90                  pass = "";
91              }
92              clone.setCredentialsProvider(
93                  new UsernamePasswordCredentialsProvider(urish.getUser(), pass));
94          }
95  
96          // Clone into a sub-directory of the working directory
97          var subdir = urish.getHumanishName();
98          if (subdir.equals("")) {
99              subdir = "repo";
100         }
101         final var cloneDir = new File(workingDirectory, subdir);
102 
103         clone.setURI(experimentDescription);
104         clone.setDirectory(cloneDir);
105         clone.setCloneSubmodules(true);
106         clone.call();
107 
108         var script = DEFAULT_SCRIPT_NAME + SYSTEM_ARG;
109         final var command = job.getCommand();
110         if (nonNull(command) && !command.isEmpty()) {
111             script = command;
112         }
113 
114         return new PyNNJobParameters(cloneDir.getAbsolutePath(),
115                 setupScript, script, job.getHardwareConfig());
116     }
117 }
