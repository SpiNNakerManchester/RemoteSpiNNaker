16  package uk.ac.manchester.cs.spinnaker.job_parameters;
17  
18  import static java.nio.charset.StandardCharsets.UTF_8;
19  
20  import java.io.File;
21  import java.io.IOException;
22  import java.io.PrintWriter;
23  
24  import uk.ac.manchester.cs.spinnaker.job.JobParameters;
25  import uk.ac.manchester.cs.spinnaker.job.nmpi.Job;
26  import uk.ac.manchester.cs.spinnaker.job.pynn.PyNNJobParameters;
27  
28  /**
29   * A {@link JobParametersFactory} that uses the {@code experimentDescription}
30   * itself as a PyNN script.
31   */
32  class DirectPyNNJobParametersFactory extends JobParametersFactory {
33      @Override
34      public JobParameters getJobParameters(final Job job,
35              final File workingDirectory, final String setupScript)
36              throws UnsupportedJobException, JobParametersFactoryException {
37          if (!job.getCode().contains("import")) {
38              throw new UnsupportedJobException();
39          }
40  
41          try {
42              return constructParameters(job, workingDirectory, setupScript);
43          } catch (final IOException e) {
44              throw new JobParametersFactoryException("Error storing script", e);
45          } catch (final Throwable e) {
46              throw new JobParametersFactoryException(
47                      "General error with PyNN Script", e);
48          }
49      }
50  
51     /**
52      * Constructs the parameters by writing the script into a local file.
53      *
54      * @param job The job to construct parameters for
55      * @param workingDirectory The directory where the job should be started
56      * @param setupScript The setup script to run
57      * @return The parameters created
58      * @throws IOException If the file can't be found to write
59      */
60      private JobParameters constructParameters(final Job job,
61              final File workingDirectory, final String setupScript)
62              throws IOException {
63          final var scriptFile = new File(workingDirectory, DEFAULT_SCRIPT_NAME);
64          try (var writer = new PrintWriter(scriptFile, UTF_8)) {
65              writer.print(job.getCode());
66          }
67  
68          return new PyNNJobParameters(workingDirectory.getAbsolutePath(),
69                  setupScript, DEFAULT_SCRIPT_NAME + SYSTEM_ARG,
70                  job.getHardwareConfig());
71      }
72  }
