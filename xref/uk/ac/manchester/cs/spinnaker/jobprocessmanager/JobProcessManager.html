92      /**
93       * A log writer that uploads to the server.
94       */
95      class UploadingJobManagerLogWriter extends JobManagerLogWriter {
96  
97          /**
98           * The timer for the interval.
99           */
100         private final Timer sendTimer;
101 
102         /**
103          * An object to synchronise on when sending data.
104          */
105         private final Integer sendSync = new Integer(0);
106 
107         /**
108          * Make a log writer that uploads the log every half second.
109          */
110         UploadingJobManagerLogWriter() {
111             sendTimer = new Timer(UPDATE_INTERVAL, new ActionListener() {
112                 @Override
113                 public void actionPerformed(final ActionEvent e) {
114                     sendLog();
115                 }
116             });
117         }
118 
119         /**
120          * Send the log now if changed.
121          */
122         private void sendLog() {
123             synchronized (sendSync) {
124                 String toWrite = null;
125                 synchronized (this) {
126                     if (isPopulated()) {
127                         toWrite = takeCache();
128                     }
129                 }
130                 if (toWrite != null && !toWrite.isEmpty()) {
131                     log("Sending cached data to job manager");
132                     jobManager.appendLog(job.getId(), toWrite);
133                 }
134             }
135         }
136 
137         @Override
138         public void append(final String logMsg) {
139             log("Process Output: " + logMsg);
140             synchronized (this) {
141                 sendTimer.restart();
142                 appendCache(logMsg);
143                 if (cacheSize() >= MAX_LOG_CACHED) {
144                     sendLog();
145                 }
146             }
147         }
148 
149         @Override
150         public void stop() {
151             synchronized (sendSync) {
152                 sendTimer.stop();
153             }
154         }
155     }

212     /**
213      * Create an object that manages the running of a single job.
214      *
215      * @param serverUrlParam
216      *            The URL to the server, used for writing back results.
217      * @param deleteOnExitParam
218      *            Whether to delete the job's resources on termination.
219      * @param isLocalParam
220      *            Whether the job is local.
221      * @param executerIdParam
222      *            The ID of the executer.
223      * @param liveUploadOutputParam
224      *            Whether to do live upload of output data.
225      * @param requestMachineParam
226      *            Whether to request a machine.
227      * @param authTokenParam
228      *            The authorisation token for the server.
229      */
230     public JobProcessManager(final String serverUrlParam,
231             final boolean deleteOnExitParam, final boolean isLocalParam,
232             final String executerIdParam, final boolean liveUploadOutputParam,
233             final boolean requestMachineParam, final String authTokenParam) {
234         this.serverUrl = requireNonNull(
235                 serverUrlParam, "--serverUrl must be specified");
236         this.executerId = requireNonNull(
237                 executerIdParam, "--executerId must be specified");
238         this.deleteOnExit = deleteOnExitParam;
239         this.isLocal = isLocalParam;
240         this.liveUploadOutput = liveUploadOutputParam;
241         this.requestMachine = requestMachineParam;
242         this.authToken = authTokenParam;
243     }

245     /**
246      * Run a single job.
247      */
248     public void runJob() {
249         try {
250             jobManager = createJobManager(serverUrl, authToken);
251 
252             // Read the job
253             job = jobManager.getNextJob(executerId);
254             projectId = new File(job.getCollabId()).getName();
255 
256             // Create a temporary location for the job
257             final File workingDirectory = createTempDir("job", ".tmp", null);
258 
259             // Download the setup script
260             String downloadUrl = serverUrl + JobManagerInterface.PATH
261                     + "/" + JobManagerInterface.SETUP_SCRIPT;
262             log("Downloading setup script from " + downloadUrl);
263             final File setupScript = downloadFile(downloadUrl,
264                     workingDirectory, JobManagerInterface.SETUP_SCRIPT);
265 
266 
267             final JobParameters parameters = getJobParameters(
268                     workingDirectory, setupScript.getAbsolutePath());
269 
270             // Create a process to process the request
271             log("Creating process from parameters");
272             final JobProcess<JobParameters> process =
273                     JOB_PROCESS_FACTORY.createProcess(parameters);
274             logWriter = getLogWriter();
275 
276             // Read the machine
277             final Machine machine = getMachine();
278 
279             // Execute the process
280             log("Running job " + job.getId() + " on " + machine + " using "
281                     + parameters.getClass() + " reporting to " + serverUrl);
282             process.execute(machine.getURL(), machine.getMachine(), parameters,
283                     logWriter);
284             logWriter.stop();
285 
286             // Get the exit status
287             processOutcome(workingDirectory, process, logWriter.getLog());
288         } catch (final Exception error) {
289             log(error);
290             reportFailure(error);
291             exit(1);
292         }
293     }
294 
295     /**
296      * Report a job failure.
297      *
298      * @param error The error of the failure.
299      */
300     private void reportFailure(final Throwable error) {
301         if ((jobManager == null) || (job == null)) {
302             log(error);
303             return;
304         }
305 
306         try {
307             String log = "";
308             if (logWriter != null) {
309                 logWriter.stop();
310                 log = logWriter.getLog();
311             }
312             String message = error.getMessage();
313             if (message == null) {
314                 message = "No Error Message";
315             }
316             jobManager.setJobError(projectId, job.getId(), message, log, "",
317                     new ArrayList<String>(), new RemoteStackTrace(error));
318         } catch (final Throwable t) {
319             // Exception while reporting exception...
320             log(t);
321             log(error);
322             exit(2);
323         }
324     }

326     /**
327      * How to run a Job Process Manager. This is the execution entry point for
328      * this Maven module.
329      *
330      * @param args
331      *            The command line arguments.
332      * @throws Exception
333      *             No guarantees made about what can go wrong.
334      */
335     public static void main(final String[] args) throws Exception {
336         String serverUrl = null;
337         boolean deleteOnExit = false;
338         boolean isLocal = false;
339         String executerId = null;
340         boolean liveUploadOutput = false;
341         boolean requestMachine = false;
342         String authToken = null;
343 
344         for (int i = 0; i < args.length; i++) {
345             switch (args[i]) {
346                 case "--serverUrl" :
347                     serverUrl = args[++i];
348                     break;
349                 case "--executerId" :
350                     executerId = args[++i];
351                     break;
352                 case "--deleteOnExit" :
353                     deleteOnExit = true;
354                     break;
355                 case "--local" :
356                     isLocal = true;
357                     break;
358                 case "--liveUploadOutput" :
359                     liveUploadOutput = true;
360                     break;
361                 case "--requestMachine" :
362                     requestMachine = true;
363                     break;
364                 case "--authToken" :
365                     try (BufferedReader r = new BufferedReader(
366                             new InputStreamReader(System.in))) {
367                         authToken = r.readLine();
368                     }
369                     break;
370                 default :
371                     throw new IllegalArgumentException(
372                             "unknown option: " + args[i]);
373             }
374         }
375 
376         new JobProcessManager(serverUrl, deleteOnExit, isLocal, executerId,
377                 liveUploadOutput, requestMachine, authToken).runJob();
378         exit(0);
379     }

395     /**
396      * Sort out the parameters to a job. Includes downloading any necessary
397      * files.
398      *
399      * @param workingDirectory
400      *            The working directory for the job, used to write files.
401      * @param setupScript
402      *            The setup script
403      * @return Description of the parameters.
404      * @throws IOException
405      *             If anything goes wrong, such as the parameters being
406      *             unreadable or the job being unsupported on the current
407      *             architectural configuration.
408      */
409     private JobParameters getJobParameters(final File workingDirectory,
410             final String setupScript) throws IOException {
411         final Map<String, JobParametersFactoryException> errors =
412                 new HashMap<>();
413         final JobParameters parameters = JobParametersFactory
414                 .getJobParameters(job, workingDirectory, setupScript, errors);
415 
416         if (parameters == null) {
417             if (!errors.isEmpty()) {
418                 throw new JobErrorsException(errors);
419             }
420             // Miscellaneous other error
421             throw new IOException(
422                     "The job did not appear to be supported on this system");
423         }
424 
425         // Get any requested input files
426         if (job.getInputData() != null) {
427             for (final DataItem input : job.getInputData()) {
428                 downloadFile(input.getUrl(), workingDirectory, null);
429             }
430         }
431 
432         return parameters;
433     }

447     /**
448      * Process the outcome of the job execution.
449      *
450      * @param workingDirectory The directory where the job was run
451      * @param process The process of the job
452      * @param log The log message of the job
453      * @throws IOException If there is an error reading or writing files
454      */
455     private void processOutcome(final File workingDirectory,
456             final JobProcess<?> process, final String log)
457             throws IOException {
458         final Status status = process.getStatus();
459         log("Process has finished with status " + status);
460 
461         final List<File> outputs = process.getOutputs();
462         final List<String> outputsAsStrings = new ArrayList<>();
463         for (final File output : outputs) {
464             if (isLocal) {
465                 outputsAsStrings.add(output.getAbsolutePath());
466             } else {
467                 try (InputStream input = new FileInputStream(output)) {
468                     jobManager.addOutput(projectId, job.getId(),
469                             output.getName(), input);
470                 }
471             }
472         }
473 
474         for (final ProvenanceItem item : process.getProvenance()) {
475             jobManager.addProvenance(
476                 job.getId(), item.getPath(), item.getValue());
477         }
478 
479         switch (status) {
480             case Error :
481                 final Throwable error = process.getError();
482                 String message = error.getMessage();
483                 if (message == null) {
484                     message = "No Error Message";
485                 }
486                 jobManager.setJobError(projectId, job.getId(), message, log,
487                         workingDirectory.getAbsolutePath(), outputsAsStrings,
488                         new RemoteStackTrace(error));
489                 break;
490             case Finished :
491                 jobManager.setJobFinished(projectId, job.getId(), log,
492                         workingDirectory.getAbsolutePath(), outputsAsStrings);
493 
494                 // Clean up
495                 process.cleanup();
496                 if (deleteOnExit) {
497                     deleteQuietly(workingDirectory);
498                 }
499                 break;
500             default :
501                 throw new IllegalStateException("Unknown status returned!");
502         }
503     }

510     /** The machine. Knows its service URL. */
511     private SpinnakerMachine machine;

515     /**
516      * Create a machine known by object.
517      *
518      * @param machineParam
519      *            The machine object.
520      */
521     Machine(final SpinnakerMachine machineParam) {
522         this.machine = machineParam;
523     }
524 
525     /**
526      * Create a machine known by service job name.
527      *
528      * @param baseUrl
529      *            The base URL for the machine.
530      * @param id
531      *            The ID for the job.
532      */
533     Machine(final String baseUrl, final int id) {
534         this.url = format("%sjob/%d/machine", baseUrl, id);
535     }
536 
537     /**
538      * Get the machine.
539      *
540      * @return the machine
541      */
542     SpinnakerMachine getMachine() {
543         return machine;
544     }
545 
546     /**
547      * Get the URL.
548      *
549      * @return the URL
550      */
551     String getURL() {
552         return url;
553     }
554 
555     @Override
556     public String toString() {
557         if (machine != null) {
558             return machine.toString();
559         }
560         return url;
561     }

564 /**
565  * How to write to the log.
566  */
567 abstract class JobManagerLogWriter implements LogWriter {
568 
569     /**
570      * Create a machine known by object.
571      *
572      * @param machine
573      *            The machine object.
574      */
575     /**
576      * The cached message.
577      */
578     private final StringBuilder cached = new StringBuilder();
579 
580 
581     /**
582      * Does the log have anything in it?
583      *
584      * @return Whether the log cache is non-empty.
585      */
586     protected synchronized boolean isPopulated() {
587         return cached.length() > 0;
588     }
589 
590     /**
591      * Adds a message to the cache.
592      *
593      * @param message The message to add
594      */
595     protected synchronized void appendCache(final String message) {
596         cached.append(message);
597     }
598 
599     /**
600      * Get the current contents of the log cache.
601      *
602      * @return Copy of the log contents.
603      */
604     public synchronized String getLog() {
605         return cached.toString();
606     }
607 
608     /**
609      * Get the current log contents and reset the internal buffer.
610      *
611      * @return The contents of the log prior to this call.
612      */
613     public final synchronized String takeCache() {
614         try {
615             return cached.toString();
616         } finally {
617             cached.setLength(0);
618         }
619     }
620 
621     /**
622      * Get the number of characters in the current log.
623      *
624      * @return The number of characters in the log.
625      */
626     public final int cacheSize() {
627         return cached.length();
628     }
629 
630     /**
631      * Stop any background activity associated with the log.
632      */
633     void stop() {
634     }
635 }

650 /**
651  * Exception indicating errors with the job factory.
652  */
653 @SuppressWarnings("serial")
654 class JobErrorsException extends IOException {
655 
656     /**
657      * The error message of the exception.
658      */
659     private static final String MAIN_MSG = "The job type was recognised"
660             + " by at least one factory, but could not be decoded.  The"
661             + " errors are as follows:";
662 
663     /**
664      * Builds an error message from a map of errors.
665      *
666      * @param errors The errors to use.
667      * @return An exception containing the errors.
668      */
669     private static String
670             buildMessage(final Map<String, ? extends Exception> errors) {
671         StringWriter buffer = new StringWriter();
672         PrintWriter bufferWriter = new PrintWriter(buffer);
673         bufferWriter.println(MAIN_MSG);
674         for (String key : errors.keySet()) {
675             bufferWriter.print(key);
676             bufferWriter.println(":");
677             errors.get(key).printStackTrace(bufferWriter);
678             bufferWriter.println();
679         }
680         return buffer.toString();
681     }
682 
683     /**
684      * Creates an exception from a set of errors.
685      *
686      * @param errors The errors to build the exception from
687      */
688     JobErrorsException(
689             final Map<String, JobParametersFactoryException> errors) {
690         super(buildMessage(errors));
691     }
