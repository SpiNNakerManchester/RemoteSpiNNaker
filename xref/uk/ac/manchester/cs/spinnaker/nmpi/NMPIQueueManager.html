50  /**
51   * Manages the NMPI queue, receiving jobs and submitting them to be run.
52   */
53  public class NMPIQueueManager {
54  
55      /**
56       * Job status when finished.
57       */
58      public static final String STATUS_FINISHED = "finished";
59  
60      /**
61       * Job status when in the queue but the executer hasn't started.
62       */
63      public static final String STATUS_QUEUED = "queued";
64  
65      /**
66       * Job status when running.
67       */
68      public static final String STATUS_RUNNING = "running";
69  
70      /**
71       * Job status when in error.
72       */
73      public static final String STATUS_ERROR = "error";
74  
75      /**
76       * The amount of time to sleep when an empty queue is detected.
77       */
78      private static final int EMPTY_QUEUE_SLEEP_MS = 10000;
79  
80      /** The queue to get jobs from. */
81      private NMPIQueue queue;
82      /** Marker to indicate if the manager is done or not. */
83      private boolean done = false;
84      /** The set of listeners for this queue. */
85      private final Set<NMPIQueueListener> listeners = new HashSet<>();
86      /** A cache of jobs that have been received. */
87      private final Map<Integer, Job> jobCache = new HashMap<>();
88      /** The log of the job so far. */
89      private final Map<Integer, NMPILog> jobLog = new HashMap<>();
90      /**
91       * Logger.
92       */
93      private static final Logger logger = getLogger(NMPIQueueManager.class);
94  
95      /** The hardware identifier for the queue. */
96      @Value("${nmpi.hardware}")
97      private String hardware;
98      /** The URL from which to load the data. */
99      @Value("${nmpi.url}")
100     private URL nmpiUrl;
101     /** The username to log in to the server with. */
102     @Value("${nmpi.username}")
103     private String nmpiUsername;
104     /** The password or API key to log in to the server with. */
105     @Value("${nmpi.password}")
106     private String nmpiPassword;
107     /**
108      * True if the password is an API key, False if the password should be used
109      * to obtain the key.
110      */
111     @Value("${nmpi.passwordIsApiKey}")
112     private boolean nmpiPasswordIsApiKey;
113 
114     /**
115      * Initialise the client.
116      */
117     @PostConstruct
118     private void initAPIClient() {
119         var apiKey = nmpiPassword;
120         if (!nmpiPasswordIsApiKey) {
121             queue = createBasicClient(nmpiUrl, nmpiUsername, nmpiPassword,
122                     NMPIQueue.class);
123             apiKey = queue.getToken(nmpiUsername).getKey();
124         }
125         queue = createApiKeyClient(nmpiUrl, nmpiUsername, apiKey,
126                 NMPIQueue.class, NMPIQueue.createProvider());
127     }
128 
129     /**
130      * Gets a job from the cache, or from the server if the job is not in the
131      * cache.
132      *
133      * @param id
134      *            The ID of the job
135      * @return The job
136      */
137     private Job getJob(final int id) {
138         synchronized (jobCache) {
139             return jobCache.computeIfAbsent(id, queue::getJob);
140         }
141     }
142 
143     /**
144      * Register a listener against the manager for new jobs.
145      *
146      * @param listener
147      *            The listener to register
148      */
149     public void addListener(final NMPIQueueListener listener) {
150         listeners.add(listener);
151     }
152 
153     public void processResponsesFromQueue() {
154         while (!done) {
155             try {
156                 // logger.debug("Getting next job");
157                 processResponse(queue.getNextJob(hardware));
158             } catch (final Exception e) {
159                 logger.error("Error in getting next job", e);
160                 sleep(EMPTY_QUEUE_SLEEP_MS);
161             }
162         }
163     }
164 
165     /**
166      * Process the response from the service.
167      *
168      * @param response The response to process
169      */
170     private void processResponse(final QueueNextResponse response) {
171         if (response instanceof QueueEmpty) {
172             sleep(EMPTY_QUEUE_SLEEP_MS);
173         } else if (response instanceof Job) {
174             processResponse((Job) response);
175         } else {
176             throw new IllegalStateException();
177         }
178     }
179 
180     /**
181      * Process the response of a Job.
182      *
183      * @param job The job to process
184      */
185     private void processResponse(final Job job) {
186         synchronized (jobCache) {
187             jobCache.put(job.getId(), job);
188         }
189         logger.debug("Job {} received", job.getId());
190         try {
191             for (final var listener : listeners) {
192                 listener.addJob(job);
193             }
194             logger.debug("Setting job status to queued");
195             job.setTimestampSubmission(
196                     job.getTimestampSubmission().withZoneRetainFields(UTC));
197             job.setTimestampCompletion(null);
198             job.setStatus(STATUS_QUEUED);
199             logger.debug("Updating job status on server");
200             queue.updateJob(job.getId(), job);
201         } catch (final IOException e) {
202             logger.error("Error in executing job", e);
203             setJobError(job.getId(), null, null, e, 0, null);
204         }
205     }
206 
207     /**
208      * Appends log messages to the log.
209      *
210      * @param id
211      *            The ID of the job
212      * @param logToAppend
213      *            The messages to append
214      */
215     public void appendJobLog(final int id, final String logToAppend) {
216         var existingLog = jobLog.computeIfAbsent(id, ignored -> new NMPILog());
217         existingLog.appendContent(logToAppend);
218         logger.debug("Job {} log is being updated", id);
219         queue.updateLog(id, existingLog);
220     }
221 
222     /**
223      * Mark a job as running.
224      *
225      * @param id
226      *            The ID of the job.
227      */
228     public void setJobRunning(final int id) {
229         logger.debug("Job {} is running", id);
230         final var job = getJob(id);
231         job.setStatus(STATUS_RUNNING);
232         logger.debug("Updating job status on server");
233         queue.updateJob(id, job);
234     }
235 
236     /**
237      * Marks a job as finished successfully.
238      *
239      * @param id
240      *            The ID of the job
241      * @param logToAppend
242      *            Any additional log messages to append to the existing log
243      *            (null if none)
244      * @param outputs
245      *            The outputs of the job (null if none)
246      * @param resourceUsage
247      *            The amount of resources used
248      * @param provenance
249      *            JSON provenance information
250      */
251     public void setJobFinished(final int id, final String logToAppend,
252             final List<DataItem> outputs, final long resourceUsage,
253             final ObjectNode provenance) {
254         logger.debug("Job {} is finished", id);
255 
256         if (nonNull(logToAppend)) {
257             appendJobLog(id, logToAppend);
258         }
259 
260         final var job = getJob(id);
261         job.setStatus(STATUS_FINISHED);
262         job.setOutputData(outputs);
263         job.setTimestampCompletion(new DateTime(UTC));
264         job.setResourceUsage(resourceUsage);
265         job.setProvenance(provenance);
266 
267         logger.debug("Updating job status on server");
268         queue.updateJob(id, job);
269 
270         jobLog.remove(id);
271         jobCache.remove(id);
272     }
273 
274     /**
275      * Marks a job as finished with an error.
276      *
277      * @param id
278      *            The ID of the job
279      * @param logToAppend
280      *            Any additional log messages to append to the existing log
281      *            (null if none)
282      * @param outputs
283      *            Any outputs generated, or null if none
284      * @param error
285      *            The error details
286      * @param resourceUsage
287      *            The amount of resources used
288      * @param provenance
289      *            JSON provenance information
290      */
291     public void setJobError(final int id, final String logToAppend,
292             final List<DataItem> outputs, final Throwable error,
293             final long resourceUsage, final ObjectNode provenance) {
294         logger.debug("Job {} finished with an error", id);
295         final var errors = new StringWriter();
296         error.printStackTrace(new PrintWriter(errors));
297         final var logMessage = new StringBuilder();
298         if (nonNull(logToAppend)) {
299             logMessage.append(logToAppend);
300         }
301         if (jobLog.containsKey(id) || (logMessage.length() > 0)) {
302             logMessage.append("\n\n==================\n");
303         }
304         logMessage.append("Error:\n");
305         logMessage.append(errors.toString());
306         appendJobLog(id, logMessage.toString());
307 
308         final var job = getJob(id);
309         job.setStatus(STATUS_ERROR);
310         job.setTimestampCompletion(new DateTime(UTC));
311         job.setOutputData(outputs);
312         job.setResourceUsage(resourceUsage);
313         job.setProvenance(provenance);
314 
315         logger.debug("Updating job on server");
316         queue.updateJob(id, job);
317 
318         jobLog.remove(id);
319         jobCache.remove(id);
320     }
321 
322     /**
323      * Close the manager.
324      */
325     public void close() {
326         done = true;
327     }
